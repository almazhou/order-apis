As a admin, I'd like to add one product to this online shop so that I can sell this product to customer.

  • Request
  • Response

Body

Type: application/json

Schema:

{
  "$schema": "http://json-schema.org/schema",
  "type": "object",
  "properties": {
    "name": {
      "type": "string"
    },
    "category":{"type":"string"},
    "description": {
      "type": "string"
    },
    "imageUrl": {
      "type": "string"
    },
    "sizes":{
      "type":"array"
    },
    "colors":{
      "type":"array"
    },
    "price":{
      "type":"object",
      "properties":{
        "amount":{
          "type":"double"
        },
        "currency":{
          "type":"string"
        }
      }
    }
  },
  "required": [
    "name",
    "description",
    "category",
    "price"
  ]
}

  "colors": ["RED","YELLOW","GREEN","PURPLE"],
